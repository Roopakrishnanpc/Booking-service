
name: Booking Service CI/CD Pipeline

on:
  push:
    branches:
      - main

env:
  IMAGE_NAME: booking-service
  DOCKER_USERNAME: ${{ secrets.DOCKER_USERNAME }}

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:

      # 1️⃣ Checkout code
      - name: Checkout Code
        uses: actions/checkout@v4

      # 2️⃣ Setup Java
      - name: Set up JDK 17
        uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '17'

      # 3️⃣ Build Spring Boot App
      - name: Build with Maven
        run: mvn clean package -DskipTests

      # 4️⃣ Login to Docker Hub
      - name: Login to Docker
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      # 5️⃣ Build Docker Image
      - name: Build Docker Image
        run: |
          docker build -t $DOCKER_USERNAME/$IMAGE_NAME:latest .

      # 6️⃣ Push Docker Image
      - name: Push Docker Image
        run: |
          docker push $DOCKER_USERNAME/$IMAGE_NAME:latest

      # 7️⃣ Setup kubectl
      - name: Setup kubectl
        uses: azure/setup-kubectl@v3

      # 8️⃣ Configure kubeconfig
      - name: Configure Kubeconfig
        run: |
          echo "${{ secrets.KUBE_CONFIG }}" | base64 --decode > kubeconfig
          export KUBECONFIG=$PWD/kubeconfig

      # 9️⃣ Deploy to Kubernetes
      - name: Deploy to K8s
        run: |
          kubectl apply -f k8s/
          kubectl rollout restart deployment booking-service
